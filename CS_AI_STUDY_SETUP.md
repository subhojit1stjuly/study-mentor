### **5. Weekly Review Script**
```powershell
# Create: review-week.ps1
cd D:\Projects\CS_AI_Study

Write-Host "`n📊 This Week's Progress`n" -ForegroundColor Cyan

# Check all active skills
Get-ChildItem .\skills -Directory | ForEach-Object {
    if (Test-Path "$_\progress.json") {
        $p = Get-Content "$_\progress.json" -Raw | ConvertFrom-Json
        if ($p.skill_status -eq "learning") {
            $name = $_.Name
            $completed = $p.modules_completed
            $total = $p.total_modules
            $pct = [math]::Round(($completed / $total) * 100, 1)
            Write-Host "  $name: $completed/$total modules ($pct%)" -ForegroundColor Green
        }
    }
}

# Count problems solved this week
Write-Host "`n💻 Problems Solved This Week:`n" -ForegroundColor Cyan
git log --since="7 days ago" --oneline --all | Select-String "LC-" | Measure-Object | 
    ForEach-Object { Write-Host "  $($_.Count) problems" -ForegroundColor Yellow }
```
